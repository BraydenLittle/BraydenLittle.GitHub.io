// Canvas Setup const canvas = document.getElementById('erasable-canvas'); const ctx = canvas.getContext('2d'); canvas.width = window.innerWidth; canvas.height = window.innerHeight; // Fill canvas with white ctx.fillStyle = 'white'; ctx.fillRect(0, 0, canvas.width, canvas.height); let isErasing = false; const eraserSize = 40; let erasedArea = 0; const requiredErasedPercent = 50; canvas.addEventListener('mousedown', () => isErasing = true); canvas.addEventListener('mouseup', () => isErasing = false); canvas.addEventListener('mousemove', erase); function erase(e) { if (!isErasing) return; // Calculate position and erase circle const rect = canvas.getBoundingClientRect(); const x = e.clientX - rect.left; const y = e.clientY - rect.top; ctx.globalCompositeOperation = 'destination-out'; ctx.beginPath(); ctx.arc(x, y, eraserSize, 0, Math.PI * 2); ctx.fill(); // Update erased area calculation updateErasedArea(); } function updateErasedArea() { const imageData = ctx.getImageData(0, 0, canvas.width, canvas.height).data; let transparentPixels = 0; for (let i = 3; i < imageData.length; i += 4) { if (imageData[i] === 0) transparentPixels++; } const erasedPercent = (transparentPixels / (canvas.width * canvas.height)) * 100; if (erasedPercent >= requiredErasedPercent) { canvas.style.display = 'none'; } } // Password Check function checkPassword() { const passwordInput = document.getElementById('password-input').value; const correctPassword = 'yourpassword'; // Replace with your actual password if (passwordInput === correctPassword) { document.getElementById('password-popup').style.display = 'none'; } else { alert('Incorrect password, try again.'); } }
